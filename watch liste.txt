    <script>
        function toggleWatchlist() {
            var content = document.querySelector('.watchlist-content');
            if (content.style.display === 'none') {
                content.style.display = 'block';
            } else {
                content.style.display = 'none';
            }
        }

        function closeWatchlist() {
            document.getElementById('watchlist').style.display = 'none';
        }

        function toggleWatchlistItems() {
            var content = document.querySelector('.watchlist-items-content');
            if (content.style.display === 'none' || content.style.display === '') {
                content.style.display = 'block';
            } else {
                content.style.display = 'none';
            }
        }

        function closeWatchlistItems() {
            document.getElementById('watchlistItems').style.display = 'none';
        }

        function addToWatchlist() {
            var indices = document.getElementById('indices').value;
            var timeFrame = document.getElementById('timeFrame').value;
            var importance = document.querySelector('input[name="importance"]:checked').value;

            var watchlistItems = document.querySelector('.watchlist-items-content');

            var itemDiv = document.createElement('div');
            itemDiv.className = 'item';

            var importanceDot = document.createElement('div');
            importanceDot.className = 'importance-dot';
            importanceDot.style.backgroundColor = importance;

            var itemContent = document.createElement('span');
            itemContent.innerHTML = indices + ' <strong>' + timeFrame + '</strong>';

            var removeButton = document.createElement('button');
            removeButton.className = 'remove-button';
            removeButton.innerHTML = '×';
            removeButton.onclick = function () {
                watchlistItems.removeChild(itemDiv);
            };

            itemDiv.appendChild(importanceDot);
            itemDiv.appendChild(itemContent);
            itemDiv.appendChild(removeButton);

            watchlistItems.appendChild(itemDiv);
        }
    </script>
